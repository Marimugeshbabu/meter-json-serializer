#include <stdio.h>
#include "json_serializer.h"

int serialize_to_json(GatewayData *g,
                      char *buf,
                      int size)
{
    int used = 0;

    used += snprintf(buf + used, size - used,
        "[{"
        "\"gatewayId\":\"%s\","
        "\"date\":\"%s\","
        "\"deviceType\":\"%s\","
        "\"interval_minutes\":%d,"
        "\"total_readings\":%d,"
        "\"values\":{"
            "\"device_count\":%d,"
            "\"readings\":[{"
                "\"media\":\"%s\","
                "\"meter\":\"%s\","
                "\"deviceId\":\"%s\","
                "\"unit\":\"%s\","
                "\"data\":[{"
                    "\"timestamp\":\"%s\","
                    "\"meter_datetime\":\"%s\","
                    "\"total_m3\":%.3f,"
                    "\"status\":\"%s\""
                "}]"
            "}]"
        "}"
        "}]",

        g->gatewayId,
        g->date,
        g->deviceType,
        g->interval_minutes,
        g->total_readings,
        g->values.device_count,
        g->values.readings[0].media,
        g->values.readings[0].meter,
        g->values.readings[0].deviceId,
        g->values.readings[0].unit,
        g->values.readings[0].data[0].timestamp,
        g->values.readings[0].data[0].meter_datetime,
        g->values.readings[0].data[0].total_m3,
        g->values.readings[0].data[0].status
    );

    if (used >= size) {
        return -1;
    }

    return used;
}
